# Create Prototype
country_event = { 
	id = nsc_events_exploration_ship.11
	title = "expcruiser.02.name"
	desc = "expcruiser.02.desc"
	picture = GFX_evt_ship_in_orbit_3
	show_sound = event_construction
	is_triggered_only = yes
	
	immediate = {
		create_leader = {
			class = scientist
			sub_type = survey
			name = random
			species = owner.species
		}
		create_fleet = {
			effect = {
				set_owner = root
				create_ship = {
					name = "Exploration Cruiser"
					design = "Prototype Exploration Cruiser"
				}
				set_location = root.capital_star.starbase
				set_leader = last_created_leader
				set_fleet_stance = evasive
			}
		}
		every_country = {
			limit = { 
				is_ai = yes
				is_country_type = default
			}
			create_leader = {
				class = scientist
				sub_type = survey
				name = random
				species = owner.species
			}
			create_fleet = {
				effect = {
					set_owner = prev
					create_ship = {
						name = "Exploration Cruiser"
						design = "Prototype Exploration Cruiser"
					}
					set_location = prev.capital_star.starbase
					set_leader = last_created_leader
					set_fleet_stance = evasive
				}
			}
		}
	}

	option = {
		name = "EXCELLENT"
	}
}

# Upgrade Project Complete
ship_event = {
	id = nsc_events_exploration_ship.13
	title = "expcruiser.04.name"
	desc = "expcruiser.04.desc"
	picture = GFX_evt_ship_in_orbit_3
	show_sound = event_construction
	is_triggered_only = yes

	immediate = {
		leader = {
			save_event_target_as = nsc_events_8_reassign_target_leader
			unassign_leader = this
		}
		create_fleet = {
			effect = {
				set_owner = root.owner
				create_ship = {
					name = "Exploration Cruiser"
					design = "Prototype Exploration Cruiser"
				}
				set_location = root.solar_system.starbase
				assign_leader = event_target:nsc_events_8_reassign_target_leader
				set_fleet_stance = evasive
			}
		}
		destroy_ship = root
	}

	option = {
		name = "EXCELLENT"
	}
}
