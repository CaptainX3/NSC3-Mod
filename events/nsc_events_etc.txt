	id = nsc_events.2
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		owner = {
			NOT = { has_modifier = nsc_headquarters_bonus }
			any_controlled_ship = { is_ship_size = headquarters }
		}
	}
	
	immediate = {
		owner = {
			add_modifier = {
				modifier = nsc_headquarters_bonus
				days = -1
			}
		}
	}
}

	picture = GFX_evt_ship_in_orbit_3
	show_sound = event_construction
	is_triggered_only = yes
	
	immediate = {
		create_leader = {
			class = scientist
			sub_type = survey
			name = random
			species = owner.species
		}
		create_fleet = {
			effect = {
				set_owner = root
				create_ship = {
					name = "Exploration Cruiser"
					design = "Prototype Exploration Cruiser"
				}
				set_location = root.capital_star.starbase
				set_leader = last_created_leader
				set_fleet_stance = evasive
			}
		}
		every_country = {
			limit = { 
				is_ai = yes
				is_country_type = default
			}
			create_leader = {
				class = scientist
				sub_type = survey
				name = random
				species = owner.species
			}
			create_fleet = {
				effect = {
					set_owner = prev
					create_ship = {
						name = "Exploration Cruiser"
						design = "Prototype Exploration Cruiser"
					}
					set_location = prev.capital_star.starbase
					set_leader = last_created_leader
					set_fleet_stance = evasive
				}
			}
		}
	}

	option = {
		name = "EXCELLENT"
	}
}

			unassign_leader = this
		}
		create_fleet = {
			effect = {
				set_owner = root.owner
				create_ship = {
					name = "Exploration Cruiser"
					design = "Prototype Exploration Cruiser"
				}
				set_location = root.solar_system.starbase
				assign_leader = event_target:nsc_events_8_reassign_target_leader
				set_fleet_stance = evasive
			}
		}
		destroy_ship = root
	}

	option = {
		name = "EXCELLENT"
	}
}

# Called when a Starbase changes owner
# THIS = Ship (Starbase)
# FROM = Former Owner (Country)
ship_event = {
	id = nsc_events.35
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		exists = starbase
	}

	immediate = {
		starbase = {
			if = {
				limit = {
					owner = {
						is_hive_empire = yes
					}
				}
				remove_starbase_building = {
					building = starbase_management
				}
				remove_starbase_building = {
					building = starbase_management_machine
				}
			}
			if = {
				limit = {
					owner = {
						is_machine_empire = yes
					}
				}
				remove_starbase_building = {
					building = starbase_management
				}
				remove_starbase_building = {
					building = starbase_management_hive
				}
			}
			if = {
				limit = {
					owner = {
						is_gestalt = no
					}
				}
				remove_starbase_building = {
					building = starbase_management_machine
				}
				remove_starbase_building = {
					building = starbase_management_hive
				}
			}
		}
	}
}
